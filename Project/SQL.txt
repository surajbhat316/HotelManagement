CREATE TABLE HOTEL(
H_ID INT PRIMARY KEY,
H_NAME VARCHAR(20),
LOCATION VARCHAR(20),
RATING INT,
MAX_NO_OF_DAYS INT
);

CREATE TABLE CUSTOMER(
C_ID INT PRIMARY KEY,
H_ID INT REFERENCES HOTEL(H_ID) ON DELETE CASCADE,
C_NAME VARCHAR(20),
AGE INT,
PHONE INT,
ADDRESS VARCHAR(30)
);


CREATE TABLE ROOMS(
R_ID INT PRIMARY KEY,
H_ID REFRENCES HOTEL(H_ID) ON DELETE CASCADE,
C_ID REFERENCES CUSTOMER(C_ID) ON DELETE CASCADE,
PRICE INT
);


CREATE TABLE CATEGORY(
R_ID REFERENCES ROOMS(R_ID) ON DELETE CASCADE,
TYPE VARCHAR(20),
CONDITONING VARCHAR(20)
);

CREATE TABLE BOOKING(
C_ID REFERENCES CUSTOMER(C_ID) ON DELETE CASCADE,
H_ID REFERENCES HOTEL(H_ID) ON DELETE CASCADE,
R_ID REFERENCES ROOMS(R_ID) ON DELETE CASCADE,
B_ID INT PRIMARY KEY,
NO_OF_DAYS INT
);


CREATE TABLE PAYMENT(
C_ID REFERENCES CUSTOMER(C_ID) ON DELETE CASCADE,
R_ID REFERENCES ROOMS(R_ID) ON DELETE CASCADE,
B_ID REFERENCES BOOKING(B_ID) ON DELETE CASCADE,
MODE VARCHAR(20)
);
